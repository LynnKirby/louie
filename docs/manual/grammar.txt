\appendix{Grammar summary}

\section{Lexical grammar}

\begin{peg}
file <- (token / trivia)*

token <-
    "elseif" !id_continue
    "else" !id_continue
    "end" !id_continue
    "false" !id_continue
    "if" !id_continue
    "print" !id_continue
    "then" !id_continue
    "true" !id_continue
    "var" !id_continue
    ID
    INT_LITERAL
    "("
    ")"
    ";"
    "=="
    "="
    "!="
    "!"
    "+"
    "-"
    "&&"
    "||"

trivia <-
    newline
    [ \t]
    "#" (!newline .)*

newline <-
    <U+000D Carriage Return> <U+000A Line Feed>?
    <U+000A Line Feed>

id_start <- [a-zA-Z_]
id_continue <- id_start / [0-9]
ID <- id_start id_continue*

INT_LITERAL <- [1-9] [0-9]* !id_continue
\end{peg}

\section{Hierarchical grammar}

\begin{cfg}
StmtSeq ::=
    Stmt*

Stmt ::=
    PrintStmt
    VarStmt
    AssignStmt
    IfStmt
    ";"

PrintStmt ::=
    "print" "(" Expr ")"

VarStmt ::=
    "var" ID ("=" Expr)?

AssignStmt ::=
    ID "=" Expr

IfStmt ::=
    "if" IfArms ("else" StmtSeq)? "end"

IfArms ::=
    Expr "then" StmtSeq ("elseif" IfArms)?

Expr ::=
    UnaryExpr
    LogicalAndExpr
    LogicalOrExpr
    UnaryExpr CompareOp UnaryExpr
    TermExpr

CompareOp ::=
    "==" | "!="

LogicalAndExpr ::=
    LogicalAndExpr "&&" UnaryExpr
    UnaryExpr "&&" UnaryExpr

LogicalOrExpr ::=
    LogicalOrExpr "||" UnaryExpr
    UnaryExpr "||" UnaryExpr

TermExpr ::=
    TermExpr ("+" | "-") UnaryExpr
    UnaryExpr ("+" | "-") UnaryExpr

UnaryExpr ::=
    PrimaryExpr
    "!" UnaryExpr
    "-" UnaryExpr

PrimaryExpr ::=
    "(" Expr ")"
    ID
    INT_LITERAL
    "true"
    "false"
\end{cfg}
